036 package com.sun.corba.ee.impl.encoding.fast.bytebuffer ;
037 
038 import java.nio.ByteBuffer ;
039 
040 /** Wrapper around a ByteBuffer that allows prepending a limited amount of data.
041  * Basic sequence of operations:
042  * <ol>
043  * <lit>new BufferWrapper
044  * <lit>buffer() to obtain buffer, which is then filled with data in any desired fashion.
045  * <lit>reset() to prepare the buffer for reading.
046  * <lit>(optional) prepend if necessary to place additional data at the start of the buffer.
047  * <lit>buffer() again to obtain the ByteBuffer to use for reading the data (prepend WILL change
048  * the ByteBuffer instance returned from buffer()!)
049  * </ol>
050  */
051 public class BufferWrapper {
052     private AllocatorBase allocator ;
053     private Slab slab ;
054     private int slabPosition ; // starting position in the Slab before this BufferWrapper was allocated.
055     private int headerSize ;
056 
057     private ByteBuffer backingStore ;
058     private ByteBuffer visible ;
059 
060     /** Allocate a buffer that contains dataSize data, and also
061      * may prepend upto headerSize bytes without extra copies.
062      */
063     public BufferWrapper( Allocator allocator, Slab slab, int headerSize, int size ) {
064         this.allocator = AllocatorBase.class.cast( allocator ) ;
065         this.headerSize = headerSize ;
066         this.slab = slab ;
067         this.slabPosition = slab.currentPosition() ;
068         backingStore = slab.allocate( headerSize + size ) ;
069         backingStore.position( headerSize ) ;
070 
071         visible = backingStore.slice() ;
072     }
073 
074     private void checkDispose() {
075         if (visible == null) 
076             throw new IllegalStateException( "BufferWrapper has already been disposed" ) ;
077     }
078 
079     /** Return the ByteBuffer managed by this BufferWrapper.
080      * Normally only used for interfacing with NIO channels.
081      */
082     public ByteBuffer buffer() {
083         checkDispose() ;
084         return visible ;
085     }
086 
087     /** Reset the buffer for reading.  Moves ByteBuffer position back to 0.
088      */
089     public ByteBuffer reset() {
090         checkDispose() ;
091         visible.position( 0 ) ;
092         return visible ;
093     }
094 
095     /** Return the amount of space remaining in the buffer.
096      */
097     public int remaining() {
098         checkDispose() ;
099         return visible.remaining() ;
100     }
101 
102     /** Prepend data from header.position() to header.limit() to the 
103      * current buffer.  This will change the value returned by buffer()!
104      * @throws IllegalArgumentException if header.limit() - header.position() 
105      * is greater than headerSize.
106      */
107     public ByteBuffer prepend( ByteBuffer header ) {
108         checkDispose() ;
109         if (header.remaining() > headerSize)
110             throw new IllegalArgumentException(
111                 "header size is greater than reserved space" ) ;
112 
113         int startPosition = backingStore.position() - header.remaining() ;
114         backingStore.position( startPosition ) ;
115         backingStore.put( header ) ;
116         backingStore.position( startPosition ) ;
117 
118         visible = backingStore.slice() ;
119         return visible ;
120     }
121 
122     /** Trim the buffer by reducing capacity to position, if possible.
123      * May return without changing capacity.  Also resets the position to 0,
124      * like reset().
125      */
126     public void trim() {
127         checkDispose() ;
128         int sizeNeeded = headerSize + visible.position() ;
129         backingStore = slab.trim( slabPosition, backingStore, sizeNeeded ) ;
130         backingStore.position( headerSize ) ;
131 
132         visible = backingStore.slice() ;
133         visible.position( 0 ) ;
134     }
135 
136     /** Notify the allocator that the space for this BufferWrapper is no longer needed.
137      * All calls to methods on a BufferWrapper will fail after a call to dispose().
138      */
139     public void dispose() {
140         checkDispose() ;
141         allocator.dispose( slab, backingStore ) ;
142         allocator = null ;
143         slab = null ;
144         backingStore = null ;
145         visible = null ;
146     }
147 }
