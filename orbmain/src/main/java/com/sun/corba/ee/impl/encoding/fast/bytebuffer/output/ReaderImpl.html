052 /** Each method reads data from the current ByteBuffer.  If not enough data
053  * is present in the current ByteBuffer, discard is called on the current
054  * ByteBuffer and we advance to the next ByteBuffer, or block if not enough 
055  * data is present.  If close() is called, all subsequent method calls will
056  * throw an IllegalStateException, and any threads blocked waiting for more
057  * data will be unblocked, and continue with an IllegalStateException from the
058  * blocking method call.
059  * <p>
060  * dataReceived and close may be safely invoked by multiple threads.
061  * The other methods must be invoked only by one thread, which is the reader of
062  * this data stream.
063  */
064 public class ReaderImpl implements Reader {
065     // Concurrency considerations:
066     // Only one thread (the consumer) may invoke the getXXX methods.
067     // dataReceived and close may be invoked by a producer thread.
068     // The consumer thread will invoke readXXX methods far more often
069     // than a typical producer will call dataReceived or (possibly) close.
070     // So buffers must be protected from concurrent access, either by locking
071     // or by a wait-free queue.  However, volatile is sufficient for current,
072     // since we just need to ensure the visibility of the value of current to
073     // all threads.
074     //
075     private BlockingQueue<BufferWrapper> buffers ;
076     private int queueSize ;
077     private volatile BufferWrapper current ;
078     private boolean closed ;
079     private long timeout ;
080     private ByteOrder byteOrder ;
081         
082     // Large enough to hold the largest primitive type.
083     public static final int HEADER_SIZE = 8 ;

085     /** Create a new ByteBufferReader.
086      * @param timeout Time in milliseconds to wait for more data:  If 0, the reader will
087      * never block.
088      * @throws TimeoutException when a read operation times out waiting for more data.
089      */
090     public ReaderImpl( long timeout ) {
091         buffers = new LinkedBlockingQueue<BufferWrapper>() ;
092         queueSize = 0 ;
093         current = null ;
094         closed = false ;
095         if (timeout < 0)
096             throw new IllegalArgumentException( "Timeout must not be negative." ) ;
097         this.timeout = timeout ;
098         this.byteOrder = ByteOrder.BIG_ENDIAN ;
099     }

113     /** Add more data for the reader thread to read.
114      * Data is assumed to be present from position() to limit().
115      */
116     public synchronized void receiveData( BufferWrapper bb ) {
117         bb.reset() ;
118         bb.buffer().order( byteOrder ) ;
119         buffers.offer( bb ) ;
120         queueSize += bb.remaining() ;
121     }

123     /** Cause all subsequent method calls on this object to throw IllegalStateException.
124      */
125     public synchronized void close() {
126         closed = true ; 
127 
128         current.dispose() ;
129         current = null ;
130 
131         for (BufferWrapper bw : buffers) {
132             bw.dispose() ;
133         }
134         buffers = null ;
135         queueSize = 0 ;
136     }

138     private synchronized int bytesAvailable() {
139         if (current == null)
140             return 0 ;
141         else 
142             return current.buffer().remaining() + queueSize ;
143     }
144 
145     // After this call, current must contain at least size bytes.
146     // This call may block until more data is available.
147     public synchronized void ensurePrimitive( int size ) {
148         if (closed)
149             throw new IllegalStateException( "ByteBufferReader is closed" ) ;
150 
151         // First ensure that there is enough space
152         while (bytesAvailable() < size) {
153             BufferWrapper next = null ;
154             if (timeout > 0) {
155                 try {
156                     next = buffers.poll( timeout, TimeUnit.MILLISECONDS ) ;
157                     if (next != null) {
158                         queueSize -= next.remaining() ;
159                     }
160                 } catch (InterruptedException exc) {
161                     throw new RuntimeException( exc ) ;
162                 }
163 
164                 if (next == null)
165                     throw new RuntimeException( "Timed out in read while waiting for more data" ) ;
166             } else {
167                 next = buffers.poll() ;
168                 if (next == null) 
169                     throw new BufferUnderflowException( ) ;
170             } 
171     
172             next.prepend( current.buffer() ) ;
173             current.dispose() ;
174             current = next ;
175         }
176     }

222     private void arraySizeCheck( int sizeInBytes ) {
223         if ((timeout == 0) && (sizeInBytes > bytesAvailable())) {
224             throw new BufferUnderflowException() ;
225         }

235     public void getByteArray( byte[] data ) {
236         arraySizeCheck( data.length ) ;
237         int offset = 0 ;
238         while (true) {
239             final ByteBuffer typedBuffer = current.buffer() ;
240             int dataSizeToRead = data.length - offset ;
241             if (dataSizeToRead > typedBuffer.remaining()) {
242                 dataSizeToRead = typedBuffer.remaining() ;
243             }
244 
245             typedBuffer.get( data, offset, dataSizeToRead ) ;
246             offset += dataSizeToRead ;
247             if (offset == data.length)
248                 break ;
249 
250             ensurePrimitive( 1 ) ;
251         }
252     }

254     public void getBytes( BufferWrapper buffer ) {
255         buffer.reset() ;
256         arraySizeCheck( buffer.remaining() ) ;
257         int offset = 0 ;
258         while (true) {
259             if (buffer.remaining() > current.remaining()) {
260                 buffer.buffer().put( current.buffer() ) ;
261             } else {
262                 int save = current.buffer().limit() ;
263                 current.buffer().limit( buffer.remaining() ) ;
264                 ByteBuffer tail = current.buffer().slice() ;
265                 current.buffer().limit( save ) ;
266                 buffer.buffer().put( tail ) ;
267                 break ;
268             }
269 
270             ensurePrimitive(1) ;
271         }
272     }
