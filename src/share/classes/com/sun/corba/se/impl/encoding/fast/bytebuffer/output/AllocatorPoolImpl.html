38 import java.io.Closeable ;
39 
40 import java.nio.ByteBuffer ;
41 
42 /** An Allocator implemented on a SlabPool.  All of the Slab 
43  * management is delegated to the SlabPool.
44  */
45 public class AllocatorPoolImpl extends AllocatorBase {
46     private SlabPoolImpl pool ;
47 
48     public AllocatorPoolImpl( int headerSize, 
49         AllocatorFactory.SlabPool pool ) {
50 
51         super( headerSize ) ;
52 
53         this.pool = SlabPoolImpl.class.cast( pool ) ;
54     }
55 
56     public int maxAllocationSize() {
57         checkClosed() ;
58         return pool.maxAllocationSize() - headerSize() ;
59     }
60 
61     public Allocator.BufferType bufferType() {
62         checkClosed() ;
63         return pool.bufferType() ;
64     }
65 
66     @Override
67     void dispose( Slab slab, ByteBuffer store ) {
68         pool.dispose( slab, store ) ;
69     }
70 
71     @Override
72     void releaseSlab( Slab slab ) {
73         pool.releaseSlab( slab ) ;
74     }
75 
76     @Override
77     Slab obtainSlab() {
78         return pool.getSlab() ;
79     }
80 }
