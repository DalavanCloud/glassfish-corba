036 package com.sun.corba.se.impl.encoding.fast.bytebuffer ;
037 
038 import java.util.Set ;
039 import java.util.HashSet ;
040 
041 import java.nio.ByteBuffer ;
042 
043 public class SlabPoolImpl implements AllocatorFactory.SlabPool {
044     private int maxAllocationSize ;
045     private long minSize ;
046     private long maxSize ;
047     private Allocator.BufferType bufferType ;
048 
049     private Set<Slab> emptySlabs ;      // Set of all empty slabs
050     private Set<Slab> fullSlabs ;       // Set of all full slabs (target of releaseSlab)
051     private Set<Slab> partialSlabs ;    // Set of all slabs in use (returned by getSlab, but not yet
052                                         // released.  State may be EMPTY or PARTIAL.
053 
054     public SlabPoolImpl( int maxAllocationSize, long minSize, long maxSize, 
055         Allocator.BufferType bufferType ) {
056 
057         this.maxAllocationSize = maxAllocationSize ;
058         this.minSize = minSize ;
059         this.maxSize = maxSize ;
060         this.bufferType = bufferType ;
061 
062         emptySlabs = new HashSet<Slab>() ;
063         fullSlabs = new HashSet<Slab>() ;
064         partialSlabs = new HashSet<Slab>() ;
065 
066         while (freeSpace() < minSize) {
067             Slab slab = new Slab( maxAllocationSize, bufferType ) ;
068             emptySlabs.add( slab ) ;
069         }
070     }
071 
072     public void close() {
073         emptySlabs = null ;
074         fullSlabs = null ;
075         partialSlabs = null ;
076     }
077 
078     public int maxAllocationSize() {
079         return maxAllocationSize ;
080     }
081 
082     public long minSize() {
083         return minSize ;
084     }
085 
086     public long maxSize() {
087         return maxSize ;
088     }
089 
090     private long setAvailableSize( Set<Slab> set ) {
091         long result = 0 ;
092         for (Slab slab : set) {
093             result += slab.sizeAvailable() ;
094         }
095         return result ;
096     }
097 
098     private long setAllocatedSize( Set<Slab> set ) {
099         long result = 0 ;
100         for (Slab slab : set) {
101             result += slab.sizeAllocated() ;
102         }
103         return result ;
104     }
105 
106     private long setDisposedSize( Set<Slab> set ) {
107         long result = 0 ;
108         for (Slab slab : set) {
109             result += slab.sizeDisposed() ;
110         }
111         return result ;
112     }
113 
114     public long freeSpace() {
115         return setAvailableSize( emptySlabs ) 
116             + setAvailableSize( partialSlabs ) ;
117     }
118 
119     public long inUseSpace() {
120         return setDisposedSize( partialSlabs ) 
121             + setDisposedSize( fullSlabs ) ;
122     }
123 
124     public long allocatedSpace() {
125         return setAllocatedSize( partialSlabs ) 
126             + setAllocatedSize( fullSlabs ) ;
127     }
128 
129     public Allocator.BufferType bufferType() {
130         return bufferType ;
131     }
132 
133     public Slab getSlab() {
134         if (emptySlabs.size() > 0) {
135             Slab anySlab = null ;
136             for (Slab slab : emptySlabs) {
137                 anySlab = slab ;
138                 break ;
139             }
140 
141             if (anySlab != null) {
142                 emptySlabs.remove( anySlab ) ;
143                 partialSlabs.add( anySlab ) ;
144             } 
145 
146             return anySlab ;
147         } else {
148             Slab result = new Slab( maxAllocationSize, bufferType ) ;
149             partialSlabs.add( result ) ;
150             return result ;
151         }
152     }
153 
154     public void releaseSlab( Slab slab ) {
155         slab.markFull() ;
156         partialSlabs.remove( slab ) ;
157         fullSlabs.add( slab ) ;
158     }
159 
160     public void dispose( Slab slab, ByteBuffer buffer ) {
161         slab.dispose( buffer ) ;
162 
163         if (slab.getState() == Slab.State.EMPTY) {
164             fullSlabs.remove( slab ) ;
165 
166             if (freeSpace() < maxSize) {
167                 emptySlabs.add( slab ) ;
168             }
169         }
170     }
171 }
