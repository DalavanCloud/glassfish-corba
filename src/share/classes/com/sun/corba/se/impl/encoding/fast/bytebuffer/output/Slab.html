036 package com.sun.corba.ee.impl.encoding.fast.bytebuffer ;
037 
038 import java.nio.ByteBuffer ;
039 
040 public class Slab {
041     public enum State { EMPTY, PARTIAL, FULL } ;
042 
043     private ByteBuffer space ;
044     private int sizeDisposed ;
045 
046     // Only for use in BufferWrapper
047     int currentPosition() {
048         return space.position() ;
049     }
050 
051     public State getState() {
052         if (sizeAllocated() == 0) {
053             return State.EMPTY ;
054         } else if (sizeAvailable() == 0) {
055             return State.FULL ;
056         } else {
057             return State.PARTIAL ;
058         }
059     }
060 
061     public int totalSize() {
062         return space.capacity() ;
063     }
064 
065     public int sizeAvailable() {
066         return space.remaining() ;
067     }
068 
069     public int sizeAllocated() {
070         return space.position() ;
071     }
072 
073     public int sizeDisposed() {
074         return sizeDisposed ;
075     }
076 
077     public void markFull() {
078         ByteBuffer result = allocate( sizeAvailable() ) ;
079         dispose( result ) ;
080     }
081 
082     public Slab( int size, Allocator.BufferType bt ) {
083         if (bt == Allocator.BufferType.DIRECT) {
084             space = ByteBuffer.allocateDirect( size ) ;
085         } else {
086             space = ByteBuffer.allocate( size ) ;
087         }
088 
089         initialize() ;
090     }
091 
092     private void initialize() {
093         space.limit( space.capacity() ) ;
094         space.position( 0 ) ;
095         sizeDisposed = 0 ;
096     }
097 
098     public ByteBuffer allocate( int size ) {
099         ByteBuffer result = null ;
100         if (size <= space.remaining()) {
101             space.limit( space.position() + size ) ;
102             result = space.slice() ;
103             space.position( space.limit() ) ;
104             space.limit( space.capacity() ) ;
105         }
106 
107         return result ;
108     }
109 
110     public void dispose( ByteBuffer buffer ) {
111         sizeDisposed += buffer.capacity() ;
112         if (getState() == State.FULL) {
113             if (sizeDisposed == sizeAllocated()) {
114                 initialize() ;
115             }
116         }
117     }
118 
119     public ByteBuffer trim( int bufferStartPosition, ByteBuffer buffer, int sizeNeeded ) {
120         ByteBuffer result = buffer ;
121 
122         // Check to see if buffer is the last buffer allocated
123         if ((bufferStartPosition + buffer.capacity()) == space.position()) {
124             space.position( bufferStartPosition ) ;
125             buffer = allocate( sizeNeeded ) ;
126         }
127 
128         return buffer ;
129     }
130 }
