36 package com.sun.corba.se.impl.encoding.fast.bytebuffer ;

38 import java.io.Closeable ;
39 
40 import java.nio.ByteBuffer ;
41 
42 /** Interface describing an allocator of BufferWrappers.  This is used to
43  * hide the complexities of direct vs. indirect ByteBuffer management.
44  * Allocator.BufferWrapper is used by the Reader and Writer interfaces.
45  * An Allocator is Closeable, and all methods on the Allocator will throw
46  * IllegalStateException after a close() call.
47  */
48 public interface Allocator extends Closeable {
49     public enum BufferType { DIRECT, HEAP } ;
50 
51     /** Return the maximum size that may be prepended to a BufferWrapper allocated
52      * by this allocator without extra copying.
53      */
54     int headerSize() ;
55 
56     /** Returns the maximum size that can be allocated by an allocate( int ) call.
57      */
58     int maxAllocationSize() ;
59 
60     /** Return the underlying buffer type used in this allocator.
61      */
62     BufferType bufferType() ;
63 
64     /** Allocate a buffer of size bytes.  Returns null if allocation is not
65      * possible. This buffer starts with position set to 0, limit set to 
66      * capacity, and mark undefined.
67      */
68     BufferWrapper allocate( int size ) ;
69 
70     /** Allocate a buffer with the largest possible size between minSize and maxSize.
71      * Returns null if allocation is not possible.  This may typically be used for allocating
72      * the largest possible buffer for a channel read operation.  Once the read completes,
73      * trim() may be called on the buffer to reclaim any unused space.  Note that the trim
74      * operation typically only succeeds if there are no intervening allocate calls between the
75      * allocate of a buffer and the trim call.
76      */
77     BufferWrapper allocate( int minSize, int maxSize ) ;
78 }
