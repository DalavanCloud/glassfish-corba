036 package com.sun.corba.se.impl.encoding.fast.bytebuffer ;
037 
038 import java.io.Closeable ;
039 
040 import java.nio.ByteOrder ;
041 
042 /** Interface that defines methods for reading primitive types and arrays
043  * of primitive types from a stream.  A stream is implemented as a sequence
044  * of BufferWrappers which are supplied by the receiveData method.
045  * The stream consumes the BufferWrappers: after all data has been read from
046  * a BufferWrapper, BufferWrapper.dispose() is called on the BufferWrapper.
047  */
048 public interface Reader extends Closeable {
049     /** If this returns true, methods will block (possibly with a timeout)
050      * until enough data is available.  If false, methods throw java.nio.BufferUnderflowException.
051      */
052     boolean isBlockingStream() ;
053 
054     /** Add more data to the stream.
055      */
056     void receiveData( BufferWrapper buffer ) ;
057 
058     /** Return the ByteOrder of the stream.
059      * All streams default to big endian byte order.
060      */
061     ByteOrder order() ;
062 
063     /** Set the ByteOrder of the stream.
064      */
065     void order( ByteOrder byteOrder ) ;
066 
067     /** Return the number of bytes available for get calls.  An attempt to 
068      * get more data than is present in the stream will either result in 
069      * blocking (if isBlocking() returns true) or a BufferUnderflowException.
070      */
071     int availableDataSize() ;
072 
073     /** Get the next boolean in the stream.  Requires 1 byte.
074      */
075     boolean getBoolean() ;
076 
077     /** Get the next byte in the stream.  Requires 1 byte.
078      */
079     byte getByte() ;
080 
081     /** Get the next character in the stream.  Requires 2 bytes.
082      */
083     char getChar() ;
084 
085     /** Get the next short in the stream.  Requires 2 bytes.
086      */
087     short getShort() ;
088 
089     /** Get the next int in the stream.  Requires 4 bytes.
090      */
091     int getInt() ;
092 
093     /** Get the next long in the stream.  Requires 8 bytes.
094      */
095     long getLong() ;
096 
097     /** Get the next float in the stream.  Requires 4 bytes.
098      */
099     float getFloat() ;
100 
101     /** Get the next double in the stream.  Requires 8 bytes.
102      */
103     double getDouble() ;
104 
105     /** Fill data with booleans (byte 1=true, 0=false) from the stream.
106      * If this method returns normally, data has been filled completely.
107      * Requires data.length bytes.
108      */
109     void getBooleanArray( boolean[] data ) ;
110 
111     /** Fill data with bytes from the stream.
112      * If this method returns normally, data has been filled completely.
113      * Requires data.length bytes.
114      */
115     void getByteArray( byte[] data ) ;
116 
117     /** Fill the buffer with data from the stream (that is, copy data
118      * from the stream to fill buffer from position to limit).
119      * This is useful when data must be read
120      * from one stream and then added to another stream for
121      * further processing.
122      */
123     void getBytes( BufferWrapper buffer ) ;
124 
125     /** Fill data with characters from the stream.
126      * If this method returns normally, data has been filled completely.
127      * Requires 2*data.length bytes.
128      */
129     void getCharArray( char[] data ) ;
130 
131     /** Fill data with characters from the stream.
132      * If this method returns normally, data has been filled completely.
133      * Requires 2*data.length bytes.
134      */
135     void getShortArray( short[] data ) ;
136 
137     /** Fill data with characters from the stream.
138      * If this method returns normally, data has been filled completely.
139      * Requires 4*data.length bytes.
140      */
141     void getIntArray( int[] data ) ;
142 
143     /** Fill data with characters from the stream.
144      * If this method returns normally, data has been filled completely.
145      * Requires 8*data.length bytes.
146      */
147     void getLongArray( long[] data ) ;
148 
149     /** Fill data with characters from the stream.
150      * If this method returns normally, data has been filled completely.
151      * Requires 4*data.length bytes.
152      */
153     void getFloatArray( float[] data ) ;
154 
155     /** Fill data with characters from the stream.
156      * If this method returns normally, data has been filled completely.
157      * Requires 8*data.length bytes.
158      */
159     void getDoubleArray( double[] data ) ;
160 }
